LEFT OUTER JOIN NAG0001.COPRDD02 AS A --By: Nairton 15/10/21
            ON  A.IDFEMP = LPAD( NF.LOJA,04,'0' )
            AND A.DATTRA = TO_CHAR( NF.DT_DOC,'YYYYMMDD' )
            AND SUBSTR( A.NUMDOC,02,08 ) = LPAD( NF.PDV,02,'0' ) || LPAD( NF.NUM_DOC,06,'0' )
LEFT OUTER JOIN NAG0001.CONOTD00 AS B --By: Nairton 15/10/21
            ON  B.IDFEMP = LPAD( NF.LOJA,04,'0' )
            AND B.DATEMS = TO_CHAR( NF.DT_DOC,'YYYYMMDD' )
            AND SUBSTR( B.NOTFIC,02,08 ) = LPAD( NF.PDV,02,'0' ) || LPAD( NF.NUM_DOC,06,'0' )
            AND B.NOTFIC = A.NUMDOC

SELECT A.IDFEMP, C.CODPRD, A.DATICS AS DATEMS, A.STARGT, A.IDFTRCFNE AS VCH_TRF, A.COTLOG AS LOGTRF
FROM COSEVD0H AS A
     INNER JOIN CODAVD05 AS B ON  ( A.IDFEMP    = B.IDFEMP    )
                              AND ( A.NUMSQCRGT = B.NUMSQCRGT )
                              AND ( A.DATICS    = B.DATEMS    )
     INNER JOIN CODAVD01 AS C ON  ( B.IDFEMP    = C.IDFEMP    )
                              AND ( B.NUMPED    = C.NUMPED    )
                              AND ( B.DATEMS    = C.DATEMS    )
WHERE 1=1
AND A.IDFEMP = '0067'
AND A.IDFTRCFNE = '7493988'
AND A.DATICS = '20211110'
